minimal doubly linked list

Below are relatively simple ways to create doubly linked lists in Python and C++.

Related: [ยง] intro to recursion with C++.

Python

With Python, it's possible to create a linked list where each node is a Python list. Then, adding nodes to the linked list is done by nesting Python lists. For example,

linked_list = [8, [3, [4, [0, []]]]]

def append(value: int, linked_list: list):
    if len(linked_list) == 0:
        linked_list.extend([value, []])
    else:
        append(value, linked_list[1])

def print_list(linked_list: list):
    if len(linked_list) == 0:
        return
    print(linked_list[0], end=" ")
    print_list(linked_list[1])

append(7, linked_list)
print_list(linked_list)  # 8 3 4 0 7 
print(linked_list)  # [8, [3, [4, [0, [7, []]]]]]

Also, Python allows its lists to be nested inside each other.

a = [1]
b = [2]
a.append(b)
b.append(a)
print(a)  # [1, [2, [...]]]
print(b)  # [2, [1, [...]]]

Using this, we can create doubly linked lists where each pair of directly linked nodes is two Python lists inside each other.

from typing import Any
from typing import Optional

class DoublyLinkedList:
    def __init__(self):
        self.head: Optional[list] = None
        self.tail: Optional[list] = None

    def append(self, value: Any):
        if self.head is None:
            self.head = [None, value, None]
            self.tail = self.head
        else:
            self.tail[2] = [self.tail, value, None]
            self.tail = self.tail[2]

    def prepend(self, value: Any):
        if self.head is None:
            self.head = [None, value, None]
            self.tail = self.head
        else:
            self.head[0] = [None, value, self.head]
            self.head = self.head[0]

    def clear(self):
        self.head = None
        self.tail = None

    def print_list(self):
        node: Optional[list] = self.head
        while node is not None:
            print(end=f"{node[1]} ")
            node = node[2]
        print()

def main():
    dll = DoublyLinkedList()
    dll.append(5)
    dll.append(3)
    dll.append(8)
    dll.append(2)
    dll.append(7)

    dll.prepend(1)

    dll.print_list()  # 1 5 3 8 2 7 
    print(dll.head)  # [None, 1, [[...], 5, [[...], 3, [[...], 8, [[...], 2, [[...], 7, None]]]]]]

if __name__ == "__main__":
    main()

C++

With C++, doubly linked lists are usually created with a Node class (or struct) containing two pointers, one pointing to the next node in the list and the other to the previous.

#include <iostream>
using namespace std;

class Node
{
public:
    int value;
    Node* previous = NULL;
    Node* next = NULL;
    Node(int value)
    {
        this->value = value;
    }
};

class DoublyLinkedList
{
public:
    void append(int value)
    {
        if (head == NULL)
        {
            head = new Node(value);
            tail = head;
        }
        else
        {
            Node* new_node_ptr = new Node(value);
            new_node_ptr->previous = tail;
            tail->next = new_node_ptr;
            tail = new_node_ptr;
        }
    }

    void prepend(int value)
    {
        if (head == NULL)
        {
            head = new Node(value);
            tail = head;
        }
        else
        {
            Node* new_node_ptr = new Node(value);
            new_node_ptr->next = head;
            head->previous = new_node_ptr;
            head = new_node_ptr;
        }
    }

    void print_list()
    {
        for (Node* ptr = head; ptr != NULL; ptr = ptr->next)
            cout << ptr->value << " ";
    }

    void clear()
    {
        Node* temp = NULL;
        while (head != NULL)
        {
            temp = head->next;
            delete head;
            head = temp;
        }
        tail = NULL;
    }

    ~DoublyLinkedList()
    {
        clear();
    }

private:
    Node* head = NULL;
    Node* tail = NULL;
};

int main()
{
    DoublyLinkedList dll;
    dll.append(5);
    dll.append(3);
    dll.append(8);
    dll.append(2);
    dll.append(7);

    dll.prepend(1);

    dll.print_list();  // 1 5 3 8 2 7 

    cout << endl;
    return 0;
}
